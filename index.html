        // Создание передачи

        function createTransfer() {

            const itemName = document.getElementById('itemName').value;

            const category = document.getElementById('itemCategory').value;

            const contactName = document.getElementById('contactName').value;

            const dueDate = document.getElementById('dueDate').value;

            const description = document.getElementById('itemDescription').value;

            if (!itemName || !category || !contactName || !dueDate) {

                tg.showPopup({

                    title: 'Ошибка',

                    message: 'Заполните все обязательные поля',

                    buttons: [{ type: 'ok' }]

                });

                return;

            }

            const newTransfer = {

                id: Date.now(),

                item: itemName,

                category: category,

                contact: contactName,

                date: new Date().toISOString().split('T')[0],

                dueDate: dueDate,

                status: 'pending',

                description: description

            };

            transfers.unshift(newTransfer);

           

            tg.showPopup({

                title: 'Успешно',

                message: `Передача "${itemName}" создана`,

                buttons: [{ type: 'ok' }]

            });

            // Имитация отправки уведомления

            tg.showAlert(`📨 Уведомление отправлено ${contactName}`);

            renderTransfers();

            renderDashboard();

            switchTab('transfers');

           

            // Очищаем форму

            document.getElementById('createTransferForm').querySelectorAll('input, textarea').forEach(field => {

                field.value = '';

            });

        }

        // Рендер категорий в выпадающий список

        function renderCategoryOptions() {

            const select = document.getElementById('itemCategory');

            select.innerHTML = '<option value="">Выберите категорию</option>';

           

            categories.forEach(category => {

                const option = document.createElement('option');

                option.value = category.name;

                option.textContent = `${category.emoji} ${category.name}`;

                select.appendChild(option);

            });

        }

        // Рендер категорий на вкладке

        function renderCategories() {

                tag.className = 'category-tag';

                tag.innerHTML = `${category.emoji} ${category.name}`;

                container.appendChild(tag);

            });

        }

        // Рендер передач

        function renderTransfers() {

            const container = document.getElementById('transfersList');

            container.innerHTML = '';

           

            if (transfers.length === 0) {

                container.innerHTML = '<div class="card"><p>Нет активных передач</p></div>';

                return;

            }

           

            transfers.forEach(transfer => {

                const card = document.createElement('div');

                card.className = 'card transfer-item';

                card.innerHTML = `

                    <div class="item-info">

                        <div class="item-name">${transfer.item}</div>

                        <div class="item-meta">

                            ${transfer.emoji || '📦'} ${transfer.category} → ${transfer.contact}<br>

                            📅 До: ${transfer.dueDate}

                        </div>

                    </div>

                    <div class="status status-${transfer.status}">

                        ${getStatusText(transfer.status)}

                    </div>

                `;

                container.appendChild(card);

            });

        }

        // Рендер дашборда

        function renderDashboard() {

            const active = transfers.filter(t => t.status === 'confirmed').length;

            const pending = transfers.filter(t => t.status === 'pending').length;

           

            document.getElementById('activeTransfers').textContent = active;

            document.getElementById('pendingTransfers').textContent = pending;

           

            // Рендер последних передач

            const recentContainer = document.getElementById('recentTransfers');

            recentContainer.innerHTML = '';

           

            const recentTransfers = transfers.slice(0, 3);

            recentTransfers.forEach(transfer => {

                const card = document.createElement('div');

                card.className = 'card transfer-item';

                card.innerHTML = `

                    <div class="item-info">

                        <div class="item-name">${transfer.item}</div>

                        <div class="item-meta">

                            ${transfer.contact} • ${transfer.dueDate}

                        </div>

                    </div>

                    <div class="status status-${transfer.status}">

                        ${getStatusText(transfer.status)}

                    </div>

                `;

                recentContainer.appendChild(card);

            });

        }

        // Получение текста статуса

        function getStatusText(status) {

            const statusMap = {

                'pending': 'Ожидает',

                'confirmed': 'Подтверждено',

                'rejected': 'Отклонено'

            };

            return statusMap[status] || status;

        }
